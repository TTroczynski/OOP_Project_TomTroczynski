114double Data::getDistance(Data& dataPoint)

116 double factor = pow(FACTOR, DECIMALPLACES);
117 double distance = sqrtl(pow(dataPoint.getX() - this->x, 2) + pow(dataPoint.getY() - this->y, 2) + pow(dataPoint.getZ() - this->z, 2));
